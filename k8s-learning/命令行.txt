1、查看异常的pod(查看状态)
kubectl get pods 

2、删除 Deployment(会自动删除相关的 Pods)
kubectl delete deployment my-llm-app

3、删除service
kubectl delete service <service-name>

4、删除 Pod(直接删除 Pod 可能还会被重建)
kubectl delete pod <pod-name>

5、删除所有相关资源
kubectl delete all -l app=<app-label>



# 3. 部署资源

kubectl get pods -l app=yolov8-triton

# 查看所有节点的污点
kubectl get nodes -o custom-columns=NAME:.metadata.name,TAINTS:.spec.taints

# 添加污点（阻止调度）
kubectl taint nodes node1 llm-exclusive=qwen3:NoSchedule

# 添加污点（尽量不调度）
kubectl taint nodes node2 gpu=true:PreferNoSchedule

# 添加污点（驱逐现有Pod）
kubectl taint nodes node3 maintenance=true:NoExecute

# 删除指定污点
kubectl taint nodes node1 llm-exclusive=qwen3:NoSchedule-

# 删除某个key的所有污点
kubectl taint nodes node1 llm-exclusive-

# 删除所有污点（不推荐）
kubectl patch node node1 -p '{"spec":{"taints":[]}}'

# 扩容命令
kubectl scale deploy vl-bundle-service-ocr-group0 --replicas=7
